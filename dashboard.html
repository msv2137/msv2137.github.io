    tmin = tmin / 10,
    tmax = tmax / 10) %>%
    mutate(prcp = prcp/10) %>%

Total Monthly Precipitation in Waikiki

weather_df %>%
  separate(date, into = c("year", "month", "day"), sep = "-") %>%
  drop_na(prcp) %>%
  filter(name == "Waikiki_HA") %>%
  mutate(month = recode_factor(month, "01" = "January", "02" = "February", "03" = "March", "04" = "April", "05" = "May", "06" = "June", "07" = "July", "08" = "August", "09" = "September", "10" = "October", "11" = "November", "12" = "December")) %>%
  group_by(name, month) %>%
  summarise(total_precip = sum(prcp)) %>%
  do(p=plot_ly(., x = ~month, y = ~total_precip, color = ~month, type = "bar", colors = "magma")) %>%
  subplot(nrows = 1, shareX = TRUE, shareY = TRUE) %>%
  layout(title = "Total Monthly Precipitation in Waikiki, HI", xaxis = list(title = "Month"), yaxis = list(title = "Precipitation (cm)"))
